```typescript
// services/llm/qwenService.ts
import Ollama from 'ollama'

class QwenService {
  private ollama: Ollama
  
  constructor() {
    this.ollama = new Ollama({ host: 'http://localhost:11434' })
  }
  
  async chat(prompt: string, model: string = 'qwen2.5:7b-instruct'): Promise<any> {
    const response = await this.ollama.chat({
      model,
      messages: [{ role: 'user', content: prompt }],
      format: 'json', // Force JSON output
      options: {
        temperature: 0.7,  // Balanced creativity
        top_p: 0.9,
        num_predict: 1000  // Max tokens
      }
    })
    
    return JSON.parse(response.message.content)
  }
  
  // Function calling support for tool use
  async chatWithTools(prompt: string, tools: Tool[]): Promise<any> {
    // Qwen 2.5 supports function calling
    return await this.ollama.chat({
      model: 'qwen2.5:7b-instruct',
      messages: [{ role: 'user', content: prompt }],
      tools: tools
    })
  }
}
```

    // Quick rule-based checks first

    if (quickChecks.reject) {
      return {
        approved: false,
        reasoning: quickChecks.reason,
        alternative: null
      }
    }
    
    // Use Qwen for complex risk assessment
    const prompt = AGENT_PROMPTS.RISK_MANAGER.assessmentTemplate(decision, portfolio)

    // Log risk decision
    await this.logRiskDecision(decision, assessment)
    
    if (!assessment.approved) {
      return {
        approved: false,
        reasoning: assessment.reasoning,
        alternative: assessment.modifications
      }
    }
    
    return {
      approved: true,
      symbol: decision.symbol,
      action: decision.action,
      positionSize: assessment.positionSize,
      leverage: assessment.leverage,
      stopLoss: assessment.stopLoss,
      takeProfit: assessment.takeProfit,
      riskAmount: assessment.riskAmount,
      riskPercent: assessment.riskPercent,
      expectedReturn: assessment.expectedReturn,
      riskRewardRatio: assessment.riskRewardRatio,
      warnings: assessment.warnings,
      timestamp: new Date().toISOString()
    }
  }
  
  private runQuickChecks(decision: FinalDecision, portfolio: Portfolio) {
    // Hard limits (no LLM needed)
    if (portfolio.currentDrawdown > 10) {
      return { reject: true, reason: 'Max drawdown exceeded' }
    }
    
    if (portfolio.openPositions.length >= 3) {
      return { reject: true, reason: 'Max concurrent positions reached' }
    }
    
    if (decision.confidence < 0.5) {
      return { reject: true, reason: 'Confidence below minimum threshold' }

    return { reject: false }

// services/agents/executionAgent.ts
export class ExecutionAgent {
  private qwen: QwenService
  private asterDex: AsterDexService
  
  async execute(trade: RiskApprovedTrade): Promise<ExecutionResult> {
    // Get current market state
    const marketState = await this.getMarketState(trade.symbol)
    
    // Ask Qwen for execution strategy
    const prompt = AGENT_PROMPTS.EXECUTION_SPECIALIST.executionTemplate({
      ...trade,
      ...marketState
    })
    
    const strategy = await this.qwen.chat(prompt)
    
    if (!strategy.readyToExecute) {
      return {
        success: false,
        reason: strategy.warnings.join(', ')
      }
    }
    
    // Execute via Aster DEX
    let result
    if (strategy.orderType === 'market') {
      result = await this.executeMarketOrder(trade)
    } else {
      result = await this.executeLimitOrder(trade, strategy.orderPrice)
    }
    
    // Log execution
    await this.logExecution(trade, strategy, result)
    
    return result
  }
  
  private async executeMarketOrder(trade: RiskApprovedTrade) {
    // Set leverage
    await this.asterDex.setLeverage(trade.symbol, trade.leverage)
    
    // Calculate quantity with 98% margin buffer
    const SAFETY_MARGIN = 0.98
    const marginToUse = trade.positionSize * SAFETY_MARGIN
    const positionValue = marginToUse * trade.leverage
    const currentPrice = await this.asterDex.getPrice(trade.symbol)
    let quantity = positionValue / currentPrice
    
    // Round to exchange precision
    const precisionInfo = await this.asterDex.getSymbolPrecision(trade.symbol)
    if (precisionInfo) {
      quantity = this.asterDex.roundQuantity(quantity, precisionInfo.quantityPrecision)
      quantity = Math.min(quantity, precisionInfo.maxQty)
    }
    
    // Place order
    const order = await this.asterDex.placeMarketOrder(
      trade.symbol,
      trade.action === 'BUY' ? 'BUY' : 'SELL',
      quantity,
      trade.leverage,
      false // not reduceOnly
    )
    
    return {
      success: true,
      orderId: order.orderId,
      executionPrice: currentPrice,
      quantity,
      slippage: 0, // Calculate actual slippage
      reasoning: 'Market order executed successfully'
    }
  }
}

export class MultiAgentOrchestrator {
  async executeTradingCycle(symbol: string): Promise<TradeResult> {
    const startTime = Date.now()
    
    // PARALLEL: Fetch all data sources simultaneously
    const [marketData, sentimentData, onchainData] = await Promise.all([
      this.getMarketData(symbol),
      this.getSentimentData(symbol),
      this.getOnChainData(symbol)
    ])
    
    // PARALLEL: All 3 analysts analyze simultaneously
    const [technical, sentiment, onchain] = await Promise.all([
      this.technicalAnalyst.analyze(symbol, marketData),
      this.sentimentAnalyst.analyze(symbol, sentimentData),
      this.onchainAnalyst.analyze(symbol, onchainData)
    ])
    
    logger.info(`⚡ Analysts completed in ${Date.now() - startTime}ms`)
    
    // SEQUENTIAL: Chief Analyst needs all reports
    const decision = await this.chiefAnalyst.synthesize(symbol, {
      technical,
      sentiment,
      onchain
    }, marketData)
    
    logger.info(`🧠 Chief Analyst decided in ${Date.now() - startTime}ms`)
    
    // SEQUENTIAL: Risk Manager needs final decision
    const riskAssessment = await this.riskManager.assessTrade(
      decision,
      await this.getPortfolio()
    )
    
    if (!riskAssessment.approved) {
      logger.warn(`🛡️ Risk Manager rejected trade: ${riskAssessment.reasoning}`)
      return { success: false, reason: riskAssessment.reasoning }
    }
    
    // SEQUENTIAL: Execute approved trade
    const result = await this.executionAgent.execute(riskAssessment)
    
    logger.info(`✅ Full cycle completed in ${Date.now() - startTime}ms`)
    
    return result
  }
}

// Cache analyst decisions for same market conditions
class DecisionCache {
  private cache = new Map<string, CachedDecision>()
  private TTL = 30_000 // 30 seconds
  
  getCached(symbol: string, marketData: MarketData): TechnicalSignal | null {
    const key = this.generateKey(symbol, marketData)
    const cached = this.cache.get(key)
    
    if (cached && Date.now() - cached.timestamp < this.TTL) {
      logger.info(`💨 Cache hit for ${symbol}`)
      return cached.signal
    }
    
    return null
  }
  
  private generateKey(symbol: string, data: MarketData): string {
    // Create cache key from rounded indicators
    return `${symbol}_${Math.round(data.rsi)}_${Math.round(data.price)}`
  }
}
